class Solution {
    public int[] twoSum(int[] nums, int target) {
        HashMapInteger, Integer box = new HashMap();
        for (int i = 0; i  nums.length; i++) {
            int need = target - nums[i];

            if (box.containsKey(need)) {
                return new int[]{ box.get(need), i };
            }

            box.put(nums[i], i);
        }

        return new int[]{};
    }
}
